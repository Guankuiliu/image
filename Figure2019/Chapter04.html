import seaborn as sns
import matplotlib.pyplot as plt
from rpy2.robjects import r, pandas2ri

pandas2ri.activate()

r('library(DAAG)') #DAAG
tinting = r('data(tinting)')
tinting = r('tinting')

sns.set_style('darkgrid')
g = sns.JointGrid('age', 'it', data = tinting)
g.plot_joint(plt.scatter, color = 'k', s = 40, edgecolor = 'r')

plt.plot(r('lowess(tinting$age,tinting$it)')[0], 
         r('lowess(tinting$age,tinting$it)')[1], 
         color = 'g', linewidth = 3)

g.ax_marg_x.boxplot(x = tinting['age'], vert = False, 
                    widths = 0.6)
g.ax_marg_y.boxplot(x = tinting['it'], widths = 0.6)

sns.rugplot(tinting['age'], color = 'darkred', axis = 'x',
            height = 0.03, alpha = 0.5)
sns.rugplot(tinting['it'], color = 'darkred', axis = 'y',
            height = 0.03, alpha = 0.5)

plt.show()
